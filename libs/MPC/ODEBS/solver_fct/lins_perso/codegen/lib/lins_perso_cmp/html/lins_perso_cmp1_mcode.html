1function p = lins_perso_cmp(A,B)
 2    
 3    nh = length(A); 
 4    C = zeros(nh);
 5    y = zeros(nh,1);
 6    p = zeros(nh,1);
 7    
 8    % cholesky
 9    for i=1:nh
10        temp = 0;
11        for k=1:i
12            temp = temp - C(k,i)*C(k,i);
13        end
14        
15        C(i,i) = sqrt(A(i,i) + temp);
16        if (C(i,i) <= 0)
17            error('error')
18        end
19
20        for j=i+1:nh
21            temp = 0;
22            for k=1:i
23                temp = temp - C(k,i)*C(k,j);
24            end
25            C(i,j) = (A(i,j)+temp)/C(i,i);
26        end
27    end
28    
29    % up and down
30    for i=1:nh
31        temp = B(i);
32        for j=1:i
33            temp = temp - C(j,i)*y(j);
34        end
35        y(i) = temp/C(i,i);
36    end
37
38    for i=nh:-1:1
39        temp = y(i);
40        for j=nh:-1:1
41            temp = temp - C(i,j)*p(j);
42        end
43        p(i) = temp/C(i,i);
44    end
45    
46end
